CREATE TABLE  "H_EMPLOYEES" 
   (	"EMPLOYEE_ID" NUMBER(6,0), 
	"F_NAME" VARCHAR2(40), 
	"L_NAME" VARCHAR2(40), 
	"JOB_DEPARTMENT" VARCHAR2(40), 
	"ADDRESS" VARCHAR2(40), 
	"CONTACT_ADD" VARCHAR2(40), 
	"USER_NAME" VARCHAR2(30), 
	"PASSWORD" VARCHAR2(30), 
	 CONSTRAINT "H_EMPLOYEES_PK" PRIMARY KEY ("EMPLOYEE_ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_H_EMPLOYEES" 
  before insert on "H_EMPLOYEES"               
  for each row  
begin   
  if :NEW."EMPLOYEE_ID" is null then 
    select "H_EMPLOYEES_SEQ".nextval into :NEW."EMPLOYEE_ID" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_H_EMPLOYEES" ENABLE;
