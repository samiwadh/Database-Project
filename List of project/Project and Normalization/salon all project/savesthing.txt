set serveroutput on
declare
cursor v1 is select c.country_name from employees e join departments d 
on 
(e.department_id=d.department_id)
join 
location l on(d.location_id=l.location_id)
join 
countries c on(l.country_id=c.country_id)where e.job_id='IT_PROG';
grop v1%rowtype;
begin 
open v1;
loop
fetch v1 into grop;
dbms_output.put_line(grop.country_name);
exit loop;
close v1;
end;
