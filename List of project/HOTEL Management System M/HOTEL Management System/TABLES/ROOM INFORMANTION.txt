   (	"ROOM_ID" NUMBER(6,0), 
	"CLASS_ID" NUMBER(9,0), 
	"DESCRIPTION" VARCHAR2(59), 
	"PRICE" NUMBER(9,0), 
	 CONSTRAINT "ROOM_INFORMATION_PK1" PRIMARY KEY ("ROOM_ID") ENABLE
   ) ;ALTER TABLE  "ROOM_INFORMATION" ADD CONSTRAINT "ROOM_INFORMATION_CON" FOREIGN KEY ("CLASS_ID")
	  REFERENCES  "ROOM_CLASS" ("CLASS_ID") ENABLE;

CREATE OR REPLACE TRIGGER  "BI_ROOM_INFORMATION" 
  before insert on "ROOM_INFORMATION"               
  for each row  
begin   
  if :NEW."ROOM_ID" is null then 
    select "ROOM_INFORMATION_SEQ1".nextval into :NEW."ROOM_ID" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_ROOM_INFORMATION" ENABLE;
