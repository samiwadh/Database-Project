CREATE TABLE  "CUSTOMER_INFORMATION" 
   (	"CUST_ID" NUMBER(8,0), 
	"CUST_FNAME" VARCHAR2(39), 
	"CUST_LNAME" VARCHAR2(30), 
	"ADDRESS" VARCHAR2(40), 
	"STATUS" VARCHAR2(30), 
	 CONSTRAINT "CUSTOMER_INFORMATION_PK" PRIMARY KEY ("CUST_ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_CUSTOMER_INFORMATION" 
  before insert on "CUSTOMER_INFORMATION"               
  for each row  
begin   
  if :NEW."CUST_ID" is null then 
    select "CUSTOMER_INFORMATION_SEQ".nextval into :NEW."CUST_ID" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_CUSTOMER_INFORMATION" ENABLE;
